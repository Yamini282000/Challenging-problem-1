\section{Question}
Show that the matrix $(t\vec{I}-\vec{n}\vec{n}^T)$ is of rank 1 for a parabola.
%
\section{Solution}
Method used : For a given nxn matrix A the rank equals the number of linearly independent rows or columns of A.

\begin{solution}
Let $\vec{n} = \myvec{a \\ b}.$
\\

\begin{align}
    \norm{\vec{n}}^2 = a^2 + b^2
    \\
    \vec{n}\vec{n}^T = \myvec{a^2 & ab\\ab & b^2}
\end{align}
\\
Now,
\begin{align}
    t &= \frac{a^2 + b^2}{e^2}
    \\
    \implies t\vec{I} &= \myvec{\frac{a^2 + b^2}{e^2} & 0 \\ 0 &  \frac{a^2 + b^2}{e^2}}
\end{align}

\begin{align}
    (t\vec{I}-\vec{n}\vec{n}^T) = \myvec{\frac{(1-e^2)a^2+b^2}{e^2} & -ab \\ -ab & \frac{a^2+(1-e^2)b^2}{e^2}}
\end{align}
For a parabola,$e$ = 1.
\\
\begin{align}
    (t\vec{I}-\vec{n}\vec{n}^T)  &= \myvec{b^2 & -ab \\ -ab & a^2}
\end{align}
Since row 2 can be expressed as a linear combination of row 1, that is
\begin{align}
 row 2 = \frac{-a}{b}\brak{row 1}   
\end{align}

Thus the number of linearly independent rows is 1 and
hence we obtain the rank of the matrix as 1.

\end{solution}
